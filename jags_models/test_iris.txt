model{

# Priors
  alpha ~ dnorm(0, 0.001)
  beta.sw ~ dnorm(0, 0.001)
  beta.pl ~ dnorm(0, 0.001)
  beta.pw ~ dnorm(0, 0.001)
  tau <- pow(sd, -2)
  sd ~ dunif(0, 1000)

  
# Likelihood
  for (i in 1:n){
    sl[i] ~ dnorm(mu[i], tau)
    mu[i] <- alpha + beta.sw * sw[i] + beta.pl * pl[i] + beta.pw * pw[i]
  }
  
# Residuals
  for (i in 1:n){
    resid[i] = sl[i] - mu[i]
  }
}
